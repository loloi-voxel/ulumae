   1: // hooks/useAutoSave.ts
   2: // Step 1.1.3: Silent auto-save with offline support
   3: // Auto-saves memorial data to Supabase every 30 seconds
   4: // Also saves on any change after a 3-second debounce (so rapid typing doesn't spam)
   5: // NEVER asks "Would you like to save?" â€” saves silently, always
   6: 
   7: import { useEffect, useRef, useCallback, useState } from 'react';
   8: import { MemorialData } from '@/types/memorial';
   9: 
  10: export type SaveStatus = 'idle' | 'saving' | 'saved' | 'error' | 'offline' | 'reconnected';
  11: 
  12: interface UseAutoSaveOptions {
  13:     memorialId: string | null;
  14:     data: MemorialData;
  15:     enabled?: boolean;         // Disable auto-save if needed
  16:     intervalMs?: number;       // Default: 30000 (30s)
  17:     debounceMs?: number;       // Default: 3000 (3s) after last change
  18: }
  19: 
  20: interface UseAutoSaveReturn {
  21:     saveStatus: SaveStatus;
  22:     lastSavedAt: Date | null;
  23:     saveNow: () => Promise<void>;   // Force immediate save
  24:     error: string | null;
  25: }
  26: 
  27: export function useAutoSave({
  28:     memorialId,
  29:     data,
  30:     enabled = true,
  31:     intervalMs = 30000,
  32:     debounceMs = 3000,
  33: }: UseAutoSaveOptions): UseAutoSaveReturn {
  34:     const [saveStatus, setSaveStatus] = useState<SaveStatus>('idle');
  35:     const [lastSavedAt, setLastSavedAt] = useState<Date | null>(null);
  36:     const [error, setError] = useState<string | null>(null);
  37: 
  38:     // Refs to track latest data without re-triggering effects
  39:     const dataRef = useRef(data);
  40:     const lastSavedDataRef = useRef<string>('');
  41:     const debounceTimerRef = useRef<NodeJS.Timeout | null>(null);
  42:     const intervalTimerRef = useRef<NodeJS.Timeout | null>(null);
  43:     const isSavingRef = useRef(false);
  44: 
  45:     // Keep data ref current
  46:     useEffect(() => {
  47:         dataRef.current = data;
  48:     }, [data]);
  49: 
  50:     // Step 1.1.3: Serialize data for save (shared between save and offline cache)
  51:     const serializeForSave = useCallback((currentData: MemorialData) => {
  52:         return {
  53:             step1: currentData.step1,
  54:             step2: currentData.step2,
  55:             step3: currentData.step3,
  56:             step4: currentData.step4,
  57:             step5: currentData.step5,
  58:             step6: currentData.step6,
  59:             step7: currentData.step7,
  60:             step8: {
  61:                 ...currentData.step8,
  62:                 coverPhoto: null,
  63:                 gallery: currentData.step8.gallery.map(g => ({
  64:                     ...g,
  65:                     file: null,
  66:                 })),
  67:                 interactiveGallery: currentData.step8.interactiveGallery?.map(ig => ({
  68:                     ...ig,
  69:                     file: null,
  70:                 })),
  71:                 voiceRecordings: currentData.step8.voiceRecordings.map(v => ({
  72:                     ...v,
  73:                     file: null,
  74:                 })),
  75:             },
  76:             step9: {
  77:                 ...currentData.step9,
  78:                 videos: currentData.step9.videos?.map(v => ({
  79:                     ...v,
  80:                     file: null,
  81:                 })),
  82:             },
  83:         };
  84:     }, []);
  85: 
  86:     // Core save function
  87:     const performSave = useCallback(async () => {
  88:         if (!memorialId || !enabled || isSavingRef.current) return;
  89: 
  90:         const currentData = dataRef.current;
  91:         const serializable = serializeForSave(currentData);
  92:         const serialized = JSON.stringify(serializable);
  93: 
  94:         if (serialized === lastSavedDataRef.current) {
  95:             return; // Nothing changed, skip
  96:         }
  97: 
  98:         // Step 1.1.3: Check if offline â€” cache locally
  99:         if (!navigator.onLine) {
 100:             try {
 101:                 localStorage.setItem(`offline-save-${memorialId}`, serialized);
 102:                 setSaveStatus('offline');
 103:             } catch {
 104:                 // localStorage full or unavailable â€” silent fail
 105:             }
 106:             return;
 107:         }
 108: 
 109:         isSavingRef.current = true;
 110:         setSaveStatus('saving');
 111:         setError(null);
 112: 
 113:         try {
 114:             const saveRes = await fetch(`/api/memorials/${memorialId}/save`, {
 115:                 method: 'POST',
 116:                 headers: { 'Content-Type': 'application/json' },
 117:                 body: JSON.stringify({
 118:                     memorialData: serializable,
 119:                 }),
 120:             });
 121: 
 122:             if (!saveRes.ok) {
 123:                 const result = await saveRes.json().catch(() => ({}));
 124:                 throw new Error(result.error || 'Save failed');
 125:             }
 126: 
 127:             lastSavedDataRef.current = serialized;
 128:             setLastSavedAt(new Date());
 129:             setSaveStatus('saved');
 130: 
 131:             // Clear any offline cache on success
 132:             localStorage.removeItem(`offline-save-${memorialId}`);
 133: 
 134:             // Reset to idle after 2 seconds (brief "Saved" then disappear)
 135:             setTimeout(() => {
 136:                 setSaveStatus(prev => prev === 'saved' ? 'idle' : prev);
 137:             }, 2000);
 138: 
 139:         } catch (err: any) {
 140:             console.error('Auto-save failed:', err);
 141:             // Fall back to offline cache
 142:             try {
 143:                 localStorage.setItem(`offline-save-${memorialId}`, serialized);
 144:             } catch { /* silent */ }
 145:             setError(err.message || 'Failed to save');
 146:             setSaveStatus('error');
 147:         } finally {
 148:             isSavingRef.current = false;
 149:         }
 150:     }, [memorialId, enabled, serializeForSave]);
 151: 
 152:     // Public "save now" function
 153:     const saveNow = useCallback(async () => {
 154:         await performSave();
 155:     }, [performSave]);
 156: 
 157:     // Debounced save on data change (3s after last change)
 158:     useEffect(() => {
 159:         if (!memorialId || !enabled) return;
 160: 
 161:         if (debounceTimerRef.current) {
 162:             clearTimeout(debounceTimerRef.current);
 163:         }
 164: 
 165:         debounceTimerRef.current = setTimeout(() => {
 166:             performSave();
 167:         }, debounceMs);
 168: 
 169:         return () => {
 170:             if (debounceTimerRef.current) {
 171:                 clearTimeout(debounceTimerRef.current);
 172:             }
 173:         };
 174:     }, [data, memorialId, enabled, debounceMs, performSave]);
 175: 
 176:     // Interval save every 30 seconds
 177:     useEffect(() => {
 178:         if (!memorialId || !enabled) return;
 179: 
 180:         intervalTimerRef.current = setInterval(() => {
 181:             performSave();
 182:         }, intervalMs);
 183: 
 184:         return () => {
 185:             if (intervalTimerRef.current) {
 186:                 clearInterval(intervalTimerRef.current);
 187:             }
 188:         };
 189:     }, [memorialId, enabled, intervalMs, performSave]);
 190: 
 191:     // Step 1.1.3: Sync offline cache when connection returns
 192:     useEffect(() => {
 193:         if (!memorialId || !enabled) return;
 194: 
 195:         const handleOnline = async () => {
 196:             const cached = localStorage.getItem(`offline-save-${memorialId}`);
 197:             if (cached) {
 198:                 setSaveStatus('reconnected');
 199:                 // Wait a beat then save
 200:                 setTimeout(async () => {
 201:                     await performSave();
 202:                 }, 1000);
 203:             }
 204:         };
 205: 
 206:         window.addEventListener('online', handleOnline);
 207:         return () => window.removeEventListener('online', handleOnline);
 208:     }, [memorialId, enabled, performSave]);
 209: 
 210:     // Save on page unload (best effort)
 211:     useEffect(() => {
 212:         if (!memorialId || !enabled) return;
 213: 
 214:         const handleBeforeUnload = () => {
 215:             const currentData = dataRef.current;
 216:             const payload = JSON.stringify({
 217:                 memorialData: serializeForSave(currentData),
 218:             });
 219:             const blob = new Blob([payload], { type: 'application/json' });
 220:             navigator.sendBeacon(`/api/memorials/${memorialId}/save`, blob);
 221:         };
 222: 
 223:         window.addEventListener('beforeunload', handleBeforeUnload);
 224:         return () => window.removeEventListener('beforeunload', handleBeforeUnload);
 225:     }, [memorialId, enabled, serializeForSave]);
 226: 
 227:     return { saveStatus, lastSavedAt, saveNow, error };
 228: }

 183: function buildStoragePath(
 184:   memorialId: string,
 185:   kind: MediaKind,
 186:   originalFileName: string

 189:   const safeName = sanitizeFileName(originalFileName);

 198: function ensureMimeAndSize(
 199:   kind: MediaKind,
 200:   mimeType: string,
 201:   byteLength: number
 202: ) {
 203:   const config = getKindConfig(kind);
 204:   const matchesMime = config.mimePrefixes.some((prefix) =>
 205:     mimeType.toLowerCase().startsWith(prefix)
 206:   );
 207: 
 208:   if (!matchesMime) {
 209:     throw new Error(`Unsupported media type for ${kind}.`);

 211: 
 212:   if (byteLength <= 0) {
 213:     throw new Error('Cannot upload an empty file.');
 214:   }

 217:     throw new Error(`File exceeds the ${Math.round(config.maxBytes / 1024 / 1024)}MB limit.`);

 221: async function uploadBufferAsMediaAsset({
 222:   admin,
 223:   memorialId,
 224:   createdBy,
 225:   kind,
 226:   buffer,
 227:   mimeType,
 228:   originalFileName,
 229:   metadata = {},
 230:   contributionId = null,
 231: }: UploadBufferInput): Promise<StoredMediaAsset> {
 232:   ensureMimeAndSize(kind, mimeType, buffer.byteLength);
 233: 
 234:   const config = getKindConfig(kind);
 235:   const storagePath = buildStoragePath(memorialId, kind, originalFileName);
 236:   const sha256Hash = crypto.createHash('sha256').update(buffer).digest('hex');
 237: 
 238:   const { error: uploadError } = await admin.storage
 239:     .from(config.bucket)
 240:     .upload(storagePath, buffer, {
 241:       contentType: mimeType,
 242:       upsert: false,
 243:     });
 244:

 246:     throw new Error(uploadError.message || 'Upload failed.');
 247:   }
 248: 
 249:   const { data: publicUrlData } = admin.storage
 250:     .from(config.bucket)
 251:     .getPublicUrl(storagePath);
 252: 
 253:   const insertPayload = {
 254:     memorial_id: memorialId,
 255:     contribution_id: contributionId,
 256:     kind,
 257:     bucket: config.bucket,
 258:     storage_path: storagePath,
 259:     public_url: publicUrlData.publicUrl,
 260:     original_file_name: originalFileName,
 261:     mime_type: mimeType,
 262:     file_size: buffer.byteLength,
 263:     sha256_hash: sha256Hash,
 264:     metadata,
 265:     created_by: createdBy,
 266:     updated_at: new Date().toISOString(),
 267:   };
 268: 
 269:   const { data, error } = await admin
 270:     .from('memorial_media_assets')
 271:     .insert(insertPayload)
 272:     .select(MEDIA_SELECT)

 276:     await admin.storage.from(config.bucket).remove([storagePath]).catch(() => undefined);

 278:   }

 414:     .from('memorial_media_assets')
 415:     .delete()
 416:     .eq('memorial_id', memorialId)
 417:     .in(
 418:       'id',
 419:       assets.map((asset) => asset.id)
 420:     );
 421: 
 422:   if (error) {
 423:     throw new Error(error.message || 'Could not delete media rows.');
 424:   }

 455: function parseSupabasePublicUrl(url: string) {
 456:   const baseUrl = process.env.NEXT_PUBLIC_SUPABASE_URL;
 457:   if (!baseUrl) return null;
 458: 
 459:   const normalizedBase = baseUrl.replace(/\/+$/, '');
 460:   const normalizedUrl = url.replace(/\/+$/, '');
 461:   const publicPrefix = `${normalizedBase}/storage/v1/object/public/`;
 462: 
 463:   if (!normalizedUrl.startsWith(publicPrefix)) {
 464:     return null;
 465:   }

 467:   const remainder = normalizedUrl.slice(publicPrefix.length);
 468:   const slashIndex = remainder.indexOf('/');
 469:   if (slashIndex === -1) return null;
 470: 
 471:   const bucket = remainder.slice(0, slashIndex) as MediaBucket;
 472:   const storagePath = remainder.slice(slashIndex + 1);
 473: 
 474:   return { bucket, storagePath };

 535: async function registerExistingStorageAsset({
 536:   admin,
 537:   memorialId,
 538:   userId,
 539:   kind,
 540:   bucket,
 541:   storagePath,
 542:   metadata = {},
 543: }: {
 544:   admin: SupabaseClient;
 545:   memorialId: string;
 546:   userId: string;
 547:   kind: MediaKind;
 548:   bucket: MediaBucket;
 549:   storagePath: string;
 550:   metadata?: Record<string, unknown>;
 551: }) {
 552:   const existingAsset = await findExistingAssetByStorageLocation(
 553:     admin,
 554:     memorialId,
 555:     bucket,
 556:     storagePath
 557:   );
 558: 
 559:   if (existingAsset) {
 560:     if (existingAsset.deletedAt) {
 561:       await restoreMemorialMediaAssets(admin, memorialId, [existingAsset.id]);
 562:       existingAsset.deletedAt = null;
 563:     }
 564:     return existingAsset;
 565:   }
 566: 
 567:   const exists = await storageObjectExists(admin, bucket, storagePath);
 568:   if (!exists) {
 569:     return null;
 570:   }
 571: 
 572:   const { data, error } = await admin.storage.from(bucket).download(storagePath);
 573:   if (error || !data) {
 574:     return null;
 575:   }
 576: 
 577:   const arrayBuffer = await data.arrayBuffer();
 578:   const buffer = Buffer.from(arrayBuffer);
 579:   const mimeType =
 580:     data.type || (bucket === 'videos' ? 'video/mp4' : 'image/jpeg');
 581:   const sha256Hash = crypto.createHash('sha256').update(buffer).digest('hex');
 582: 
 583:   const { data: publicUrlData } = admin.storage.from(bucket).getPublicUrl(storagePath);
 584: 
 585:   const insertPayload = {
 586:     memorial_id: memorialId,
 587:     kind,
 588:     bucket,
 589:     storage_path: storagePath,
 590:     public_url: publicUrlData.publicUrl,
 591:     original_file_name: splitStoragePath(storagePath).fileName,
 592:     mime_type: mimeType,
 593:     file_size: buffer.byteLength,
 594:     sha256_hash: sha256Hash,
 595:     metadata,
 596:     created_by: userId,
 597:     updated_at: new Date().toISOString(),
 598:   };
 599: 
 600:   const { data: inserted, error: insertError } = await admin
 601:     .from('memorial_media_assets')
 602:     .insert(insertPayload)
 603:     .select(MEDIA_SELECT)
 604:     .single();
 605: 
 606:   if (insertError || !inserted) {
 607:     throw new Error(insertError?.message || 'Could not register legacy media.');
 608:   }
 609: 
 610:   return serializeMediaAsset(inserted as MediaAssetRow);
 611: }
